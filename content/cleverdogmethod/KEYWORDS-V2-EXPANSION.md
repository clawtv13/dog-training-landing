KEYWORD_COOLDOWN_DAYS = 30

def filter_recent_keywords(keywords, published_posts):
    """Remove keywords published within cooldown period"""
    cutoff_date = datetime.now() - timedelta(days=KEYWORD_COOLDOWN_DAYS)
    recent_slugs = {
        p['slug'] for p in published_posts 
        if datetime.fromisoformat(p['published']) > cutoff_date
    }
    return [kw for kw in keywords if kw['slug'] not in recent_slugs]
